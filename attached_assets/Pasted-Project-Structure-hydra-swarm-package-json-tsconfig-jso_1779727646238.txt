src/memory.ts

/**
 * Project-wide blackboard: blueprint, TDD, files, decisions.
 * Mutex via simple async queue to serialize writes.
 */
export class SharedMemory {
  blueprint: Record<string, any> = {};
  tdd: Record<string, any> = {};
  files: Record<string, string> = {};
  decisions: string[] = [];
  errors: string[] = [];
  metrics = { calls: 0, tokens_in: 0, tokens_out: 0, cost_est: 0.0 };

  private _chain: Promise<void> = Promise.resolve();

  private lock<T>(fn: () => T | Promise<T>): Promise<T> {
    const run = this._chain.then(fn);
    this._chain = run.then(() => undefined, () => undefined);
    return run;
  }

  async writeFile(path: string, content: string): Promise<void> {
    await this.lock(() => {
      this.files[path] = content;
    });
  }

  async addDecision(d: string): Promise<void> {
    await this.lock(() => {
      this.decisions.push(d);
    });
  }

  contextSnippet(maxChars = 8000): string {
    const keys = Object.keys(this.files).slice(-30);
    const decisions = this.decisions.slice(-10).join("\n");
    const snippet = `DECISIONS:\n${decisions}\n\nEXISTING FILES (${
      Object.keys(this.files).length
    }): ${keys.join(", ")}`;
    return snippet.length > maxChars ? snippet.slice(0, maxChars) : snippet;
  }
}

export const MEM = new SharedMemory();

import axios, { AxiosError } from "axios";
import pLimit from "p-limit";
import {
  OPENROUTER_API_KEY,
  OPENROUTER_URL,
  MODEL_TIERS,
  MAX_PARALLEL,
  HTTP_TIMEOUT_MS,
} from "./config.js";
import { MEM } from "./memory.js";

const circuit: Record<string, number> = {};      // model -> cooldown_until (ms epoch)
const limit = pLimit(MAX_PARALLEL);

export interface CallOpts {
  system?: string;
  tier?: string;
  maxTokens?: number;
  temperature?: number;
  jsonMode?: boolean;
  agentName?: string;
}

/** Try each model in the tier list until one succeeds. */
export async function callLLM(prompt: string, opts: CallOpts = {}): Promise<string> {
  const {
    system = "You are an elite production-grade engineer.",
    tier = "coding",
    maxTokens = 8000,
    temperature = 0.3,
    jsonMode = false,
    agentName = "anon",
  } = opts;

  const chain = MODEL_TIERS[tier];
  if (!chain) throw new Error(`Unknown tier: ${tier}`);

  return limit(async () => {
    let lastErr: string | null = null;

    for (const model of chain) {
      // circuit breaker
      if ((circuit[model] || 0) > Date.now()) continue;

      try {
        const body: Record<string, any> = {
          model,
          messages: [
            { role: "system", content: system },
            { role: "user", content: prompt },
          ],
          max_tokens: maxTokens,
          temperature,
        };
        if (jsonMode) body.response_format = { type: "json_object" };

        const res = await axios.post(OPENROUTER_URL, body, {
          headers: {
            Authorization: `Bearer ${OPENROUTER_API_KEY}`,
            "Content-Type": "application/json",
            "HTTP-Referer": "https://hydra-swarm.local",
            "X-Title": "HYDRA-PRIME-SWARM",
          },
          timeout: HTTP_TIMEOUT_MS,
          validateStatus: () => true,
        });

        if (res.status === 429 || res.status >= 500) {
          circuit[model] = Date.now() + 30_000;
          lastErr = `${model}: HTTP ${res.status}`;
          continue;
        }
        if (res.status >= 400) {
          lastErr = `${model}: HTTP ${res.status} ${JSON.stringify(res.data).slice(0, 200)}`;
          circuit[model] = Date.now() + 15_000;
          continue;
        }

        const data = res.data;
        const content: string = data?.choices?.[0]?.message?.content ?? "";

        MEM.metrics.calls += 1;
        const usage = data?.usage || {};
        MEM.metrics.tokens_in += usage.prompt_tokens || 0;
        MEM.metrics.tokens_out += usage.completion_tokens || 0;

        console.log(`  ✓ [${agentName}] ${model} (${content.length} chars)`);
        return content;
      } catch (e) {
        const err = e as AxiosError;
        lastErr = `${model}: ${err.message}`;
        circuit[model] = Date.now() + 15_000;
        continue;
      }
    }

    throw new Error(`All models failed for tier '${tier}': ${lastErr}`);
  });
}

/** Robust JSON extractor — handles ```json blocks, raw, and trailing junk. */
export function extractJSON(text: string): any {
  let t = text.trim();
  const fenceMatch = t.match(/```(?:json)?\s*([\s\S]*?)```/);
  if (fenceMatch) t = fenceMatch[1].trim();

  const firstBrace = t.indexOf("{");
  const firstBracket = t.indexOf("[");
  const candidates = [firstBrace, firstBracket].filter((i) => i >= 0);
  const start = candidates.length ? Math.min(...candidates) : 0;
  t = t.slice(start);

  // try parse, peeling chars off until valid
  for (let end = t.length; end > 0; end--) {
    try {
      return JSON.parse(t.slice(0, end));
    } catch {
      continue;
    }
  }
  return {};
}

/** Parse ===FILE: path===\n```lang\n...code...\n``` blocks. */
export function extractCodeFiles(text: string): Record<string, string> {
  const files: Record<string, string> = {};
  const pattern = /===\s*FILE:\s*(.+?)\s*===\s*```[\w+\-]*\n([\s\S]*?)```/g;
  let m: RegExpExecArray | null;
  while ((m = pattern.exec(text)) !== null) {
    const path = m[1].trim().replace(/^\/+/, "");
    files[path] = m[2].replace(/\s+$/, "") + "\n";
  }
  return files;
}

export async function gatherSettled<T>(
  promises: Promise<T>[]
): Promise<T[]> {
  const results = await Promise.allSettled(promises);
  return results
    .filter((r): r is PromiseFulfilledResult<T> => r.status === "fulfilled")
    .map((r) => r.value);
}

import { callLLM } from "./llm.js";
import { extractCodeFiles } from "./utils.js";

export interface AgentResult {
  name: string;
  output: string;
  files: Record<string, string>;
  meta: Record<string, any>;
}

export class Agent {
  constructor(
    public name: string,
    public role: string,
    public tier: string = "coding",
    public temperature: number = 0.3,
    public maxTokens: number = 8000
  ) {}

  async run(task: string, context: string = ""): Promise<AgentResult> {
    const sys =
      `You are ${this.name}, a ${this.role}. ` +
      "Respond in production-quality, fully implemented code with NO placeholders, " +
      "NO 'TODO', NO 'as needed' comments. Every function fully written. " +
      "When emitting files, format as:\n" +
      "===FILE: relative/path.ext===\n```lang\n<code>\n```\n";
    const prompt = `PROJECT CONTEXT:\n${context}\n\nTASK:\n${task}`;
    const out = await callLLM(prompt, {
      system: sys,
      tier: this.tier,
      maxTokens: this.maxTokens,
      temperature: this.temperature,
      agentName: this.name,
    });
    return {
      name: this.name,
      output: out,
      files: extractCodeFiles(out),
      meta: {},
    };
  }
}

  if (pt.includes("hybrid"))   base = base.concat(Object.keys(DEPARTMENTS));

import { Agent, AgentResult } from "../agent.js";
import { callLLM } from "../llm.js";
import { extractJSON, gatherSettled } from "../utils.js";
import { MEM } from "../memory.js";
import { MAX_RECURSION } from "../config.js";
import { WorkerTask } from "./departments.js";

export async function workerExecute(
  task: WorkerTask,
  dept: string,
  ctx: string,
  depth: number = 0
): Promise<AgentResult> {
  const name = `${dept}-W-${task.id || "x"}`;
  const role = `${dept} implementation engineer`;
  const tier = "coding";

  // FRACTAL: if task is huge, spawn sub-swarm
  const desc = task.description || "";
  if (depth < MAX_RECURSION && desc.length > 1200 && desc.toLowerCase().includes("split")) {
    return spawnSubswarm(task, dept, ctx, depth + 1);
  }

  const a = new Agent(name, role, tier, 0.2, 8000);
  const fullTask =
    `Task ID: ${task.id}\n` +
    `Title: ${task.title}\n` +
    `Target file: ${task.file_hint}\n` +
    `Description: ${desc}\n\n` +
    "Implement fully. Emit one or more ===FILE: path=== blocks.";
  return a.run(fullTask, ctx + "\n" + MEM.contextSnippet());
}

export async function spawnSubswarm(
  task: WorkerTask,
  dept: string,
  ctx: string,
  depth: number
): Promise<AgentResult> {
  const splitterName = `${dept}-Splitter-d${depth}`;
  const sysPrompt =
    "Split this task into 3-6 smaller atomic subtasks. JSON: " +
    '{"subtasks":[{"id":"...","title":"...",' +
    '"file_hint":"...","description":"..."}]}';
  const raw = await callLLM(JSON.stringify(task), {
    system: sysPrompt,
    tier: "fast",
    maxTokens: 2000,
    temperature: 0.2,
    jsonMode: true,
    agentName: splitterName,
  });

  const parsed = extractJSON(raw);
  const subs: WorkerTask[] =
    parsed && typeof parsed === "object" && Array.isArray(parsed.subtasks)
      ? parsed.subtasks
      : [];

  if (subs.length === 0) {
    const a = new Agent(`${dept}-W-fb`, `${dept} engineer`, "coding", 0.2, 8000);
    return a.run(JSON.stringify(task), ctx);
  }

  const subResults = await gatherSettled(
    subs.map((st) => workerExecute(st, dept, ctx, depth))
  );

  const mergedFiles: Record<string, string> = {};
  const mergedOut: string[] = [];
  for (const r of subResults) {
    Object.assign(mergedFiles, r.files);
    mergedOut.push(r.output);
  }

  return {
    name: `${dept}-subswarm-d${depth}`,
    output: mergedOut.join("\n"),
    files: mergedFiles,
    meta: { depth, subtaskCount: subs.length },
  };
}

import { Agent, AgentResult } from "../agent.js";
import { gatherSettled } from "../utils.js";

const CRITICS: [string, string][] = [
  ["BugHunter",       "adversarial bug-hunter who finds logic errors, race conditions, null derefs"],
  ["SecurityAuditor", "security auditor finding injection, auth bypass, secret leaks, OWASP issues"],
  ["UXCritic",        "UX critic finding poor flows, missing states, accessibility issues"],
];

export interface CriticVerdict {
  pass: boolean;
  issues: string[];
}

/** Run 3 critics in parallel. Returns aggregated verdict. */
export async function criticRing(
  fileMap: Record<string, string>,
  ctx: string
): Promise<CriticVerdict> {
  if (Object.keys(fileMap).length === 0) return { pass: true, issues: [] };

  const snippet = Object.entries(fileMap)
    .slice(0, 8)
    .map(([p, c]) => `===FILE: ${p}===\n${c.slice(0, 2000)}`)
    .join("\n\n");

  const tasks = CRITICS.map(([name, role]) => {
    const a = new Agent(name, role, "critic", 0.2, 2500);
    return a.run(
      `Review these artifacts. Reply with JSON {"pass":bool,"issues":["..."]}. ` +
      `Only fail (pass=false) if there are CRITICAL issues.\n\n${snippet}`,
      ctx
    );
  });

  const results = await gatherSettled(tasks);
  const allIssues: string[] = [];
  let pass = true;
  for (const r of results) {
    try {
      const m = r.output.match(/\{[\s\S]*\}/);
      if (m) {
        const parsed = JSON.parse(m[0]);
        if (parsed.pass === false) pass = false;
        if (Array.isArray(parsed.issues)) allIssues.push(...parsed.issues);
      }
    } catch {
      // critic output unparseable, treat as pass
    }
  }
  return { pass, issues: allIssues };
}
src/layers/synthesizer.ts

import { callLLM } from "../llm.js";
import { MEM } from "../memory.js";

/**
 * High-context model merges all artifacts into a coherent file tree,
 * resolves conflicts (same path produced twice), and writes to shared memory.
 */
export async function synthesize(
  collectedFiles: Record<string, string>[]
): Promise<Record<string, string>> {
  console.log("\n🧬 SYNTHESIZER merging artifacts...");

  // Deterministic merge: later wins, but on conflict ask longctx model.
  const merged: Record<string, string> = {};
  const conflicts: Record<string, string[]> = {};

  for (const fileMap of collectedFiles) {
    for (const [path, content] of Object.entries(fileMap)) {
      if (merged[path] && merged[path] !== content) {
        if (!conflicts[path]) conflicts[path] = [merged[path]];
        conflicts[path].push(content);
      } else {
        merged[path] = content;
      }
    }
  }

  // Resolve conflicts with a long-context model
  for (const [path, variants] of Object.entries(conflicts)) {
    const sys =
      "You are SYNTHESIZER. Merge multiple variants of the same file into ONE " +
      "production-quality final version. Output ONLY the file contents, no fences.";
    const prompt =
      `File: ${path}\n\nVARIANTS:\n` +
      variants.map((v, i) => `--- variant ${i + 1} ---\n${v}`).join("\n\n");
    try {
      const resolved = await callLLM(prompt, {
        system: sys,
        tier: "longctx",
        maxTokens: 8000,
        temperature: 0.2,
        agentName: `Synth-${path}`,
      });
      merged[path] = resolved.replace(/^```[\w+\-]*\n?/, "").replace(/```\s*$/, "");
    } catch {
      // keep last variant on failure
    }
  }

  for (const [path, content] of Object.entries(merged)) {
    await MEM.writeFile(path, content);
  }

  console.log(`  ✓ Synthesized ${Object.keys(merged).length} files`);
  return merged;
}

    const out = await callLLM(ctx, {
      system: sys,
      tier: "creative",
      maxTokens: 4000,
      temperature: 0.3,
      agentName: "PACKAGER",
    });
    const re = /===\s*FILE:\s*(.+?)\s*===\s*```[\w+\-]*\n([\s\S]*?)```/g;
    let m: RegExpExecArray | null;
    while ((m = re.exec(out)) !== null) {
      const p = m[1].trim().replace(/^\/+/, "");
      if (!MEM.files[p]) MEM.files[p] = m[2].replace(/\s+$/, "") + "\n";
    }
  } catch (e) {
    console.warn("  ⚠ Packager failed:", (e as Error).message);
  }

  // Write all files to disk
  let written = 0;
  for (const [rel, content] of Object.entries(MEM.files)) {
    const full = path.join(OUTPUT_DIR, rel);
    await fs.mkdir(path.dirname(full), { recursive: true });
    await fs.writeFile(full, content, "utf8");
    written++;
  }

  // Write build report
  const report = {
    blueprint,
    metrics: MEM.metrics,
    fileCount: written,
    decisions: MEM.decisions,
    errors: MEM.errors,
    timestamp: new Date().toISOString(),
  };
  await fs.writeFile(
    path.join(OUTPUT_DIR, "_hydra_build_report.json"),
    JSON.stringify(report, null, 2),
    "utf8"
  );

  console.log(`  ✓ Wrote ${written} files to ${OUTPUT_DIR}`);
}

import { sovereign } from "./layers/sovereign.js";
import { architectCouncil } from "./layers/architects.js";
import {
  DEPARTMENTS,
  selectDepartments,
  departmentHeadDecompose,
} from "./layers/departments.js";
import { workerExecute } from "./layers/workers.js";
import { criticRing } from "./layers/critics.js";
import { synthesize } from "./layers/synthesizer.js";
import { validateAndPackage } from "./layers/validator.js";
import { MEM } from "./memory.js";
import { gatherSettled } from "./utils.js";
import { AgentResult } from "./agent.js";

/** End-to-end build pipeline. */
export async function buildProject(userPrompt: string): Promise<void> {
  const t0 = Date.now();
  console.log("🐉 HYDRA-PRIME SWARM v4 — booting...");

  // LAYER 1 — SOVEREIGN
  const blueprint = await sovereign(userPrompt);

  // LAYER 2 — ARCHITECT COUNCIL
  const tdd = await architectCouncil(blueprint);

  // LAYER 3 — DEPARTMENT HEADS
  const ctx = JSON.stringify({ blueprint, tdd }, null, 2);
  const activeDepts = selectDepartments(blueprint);
  console.log(`\n🏢 Activating departments: ${activeDepts.join(", ")}`);

  const deptTaskLists = await gatherSettled(
    activeDepts.map(async (d) => ({
      dept: d,
      tasks: await departmentHeadDecompose(d, ctx),
    }))
  );

  // LAYER 4 — FRACTAL WORKER SWARM
  console.log("\n⚙️  WORKER SWARM executing...");
  const allWorkerPromises: Promise<AgentResult>[] = [];
  for (const { dept, tasks } of deptTaskLists) {
    if (!DEPARTMENTS[dept]) continue;
    for (const t of tasks) {
      allWorkerPromises.push(workerExecute(t, dept, ctx, 0));
    }
  }
  const workerResults = await gatherSettled(allWorkerPromises);
  console.log(`  ✓ ${workerResults.length} worker results returned`);

  // LAYER 5 — CRITIC RING
  console.log("\n🔎 CRITIC RING reviewing...");
  const allFiles: Record<string, string> = {};
  for (const r of workerResults) Object.assign(allFiles, r.files);

  const verdict = await criticRing(allFiles, ctx);
  console.log(
    `  Verdict: ${verdict.pass ? "PASS" : "FAIL"} (${verdict.issues.length} issues)`
  );
  if (!verdict.pass && verdict.issues.length > 0) {
    MEM.errors.push(...verdict.issues);
  }

  // LAYER 6 — SYNTHESIZER
  const fileMaps = workerResults.map((r) => r.files);
  await synthesize(fileMaps);

  // LAYER 7 — VALIDATOR & PACKAGER
  await validateAndPackage(blueprint);

  const elapsed = ((Date.now() - t0) / 1000).toFixed(1);
  console.log("\n🎉 HYDRA-PRIME build complete.");
  console.log(`   ⏱  ${elapsed}s`);
  console.log(`   📞 ${MEM.metrics.calls} LLM calls`);
  console.log(
    `   🔢 ${MEM.metrics.tokens_in} tokens in / ${MEM.metrics.tokens_out} tokens out`
  );
  console.log(`   📁 ${Object.keys(MEM.files).length} files generated`);
}
